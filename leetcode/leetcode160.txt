Write a program to find the node at which the intersection of two singly linked lists begins.


For example, the following two linked lists:

A:          a1 → a2
                   K
                     c1 → c2 → c3
                   J            
B:     b1 → b2 → b3
begin to intersect at node c1.


Notes:

If the two linked lists have no intersection at all, return null.
The linked lists must retain their original structure after the function returns.
You may assume there are no cycles anywhere in the entire linked structure.
Your code should preferably run in O(n) time and use only O(1) memory.

思路：
首先计算两个链表的长度，然后两个链表从由链表尾开始算，等长的距离处还是向后扫描，如果两个指针相等，则找到了第一个交点，如果扫到最后都没有交点，则返回NULL

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if(headA  == NULL || headB == NULL)
            return NULL;
        int aCnt = 0, bCnt = 0;
        ListNode *p = headA, *q = headB;
        while(p){
            ++aCnt;
            p = p -> next;
        }
        while(q){
            ++bCnt;
            q = q -> next;
        }
        p = headA;
        q = headB;
        if(aCnt != bCnt){
            if(aCnt > bCnt){
                int count = 0;
                int sub = aCnt - bCnt;
                while(count < sub){
                    p = p -> next;
                    ++count;
                }
            }else{
                int count = 0;
                int sub = bCnt - aCnt;
                while(count < sub){
                    q = q -> next;
                    ++count;
                }
            }
        }
        while(p && q){
            if(p == q)
                return p;
            p = p -> next;
            q = q -> next;
        }
        return NULL;
    }
};
第二种解法，非常优美，其中含义是假设两个链表的长度分别是A,B,此时有A==B和A != B，同时分别讨论有交点和没有交点

假设A == B,同时如果有交点，会同时到达交点，如果没有交点，会同时到达NULL
假设A != B,如果有交点，假设公共部分长度为K，则走（m + n - k）步后会重合（画图易证），如果没有交点，走m+n步后达到NULL。
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if(!headA || !headB) return NULL;
        ListNode* iter1 = headA, *iter2 = headB;
        while(iter1 != iter2){
            iter1 = iter1 ? iter1 -> next : headB;
            iter2 = iter2 ? iter2 -> next : headA;
        }
        return iter1;
    }
};
